/*
 * Complete the 'handshake' function below.
 *
 * The function is expected to return an INTEGER.
 * The function accepts INTEGER n as parameter.
 */
solution ----------------------------
int handshake(int n) {
    if (n <= 1) {
        return 0;
     }

    return (n * (n - 1)) / 2;
}
